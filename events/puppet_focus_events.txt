country_event = {
    id = 770102

    is_triggered_only = yes

    title = "Focus: Rare Materials"
    title = "Focus: Rare Materials"

    option = {
        name = "Player should not see this"
        add_country_modifier = { #set cooldown modifier for puppet
            name = "puppet_focus_set"
            duration = 365
        }
        clr_country_flag = puppet_focus_rares #flags for LUA
        clr_country_flag = puppet_focus_energy
        clr_country_flag = puppet_focus_metal
        clr_country_flag = puppet_focus_navy
        clr_country_flag = puppet_focus_air
        clr_country_flag = puppet_focus_army
        clr_country_flag = puppet_focus_oil

        set_country_flag = puppet_focus_rares

        FROM = { #reset cooldown modifier for master, to keep all cooldowns in sync
            remove_country_modifier = puppet_focus_set
            add_country_modifier = {
                name = "puppet_focus_set"
                duration = 365
            }
        }

        any_country = { #reset cooldown modifiers for all other puppets, to keep all cooldowns in sync
            limit = {
                ai = yes
                vassal_of = FROM
                has_country_modifier = puppet_focus_set
            }
            remove_country_modifier = puppet_focus_set
            add_country_modifier = {
                name = "puppet_focus_set"
                duration = 365
            }
        }

        random_country = { # horrible hack to remove cooldown modifier from master if any other puppet DOESNT have the cooldown modifier
            limit = {
                ai = yes
                vassal_of = FROM
                not = { has_country_modifier = puppet_focus_set }
            }
            FROM = {
                remove_country_modifier = puppet_focus_set
            }
        }
        ai_chance = { factor = 100 }
    }
}
