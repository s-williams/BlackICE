country_event = {

	id = 50109
	
	trigger = {
		#any_owned_province = { strategic_resource = airportbase }
		not = { has_country_modifier = airportbase_build_1 }
		not = { has_country_modifier = delay_airportbase_build_1 }
		Hangar_Maintenance = 1
	}

	mean_time_to_happen = {
		days = 1
	}
	
	title = "Expand An Airbase"
	desc = "EVTDESC50109"
	picture = "expanding-airbase"
		
	option = {
		name = "We Need This"
		
		random_owned = {
			limit = {
				has_province_flag = airbase_point
				air_base = 1
				not = { air_base = 7 }
				#has_province_flag = majorairbase
				controlled_by = THIS
			}
			air_base = 4
		}
		add_country_modifier = {
			name = "airportbase_build_1"
			duration = 120
		}
		ai_chance = { factor = 100 }
	}
	
	option = {
		name = "No Need"
		add_country_modifier = {
			name = "delay_airportbase_build_1"
			duration = 120
		}
		ai_chance = { factor = 0 }
	}
}
