country_event = {
	
	id = 95252
	is_triggered_only = yes
	major = yes

	title = "EVTNAME508"
	desc = "EVTDESC508"
	picture = "ger_surrender"

	option = {
		name = "EVTOPTA511"
		ai_chance = {
		  factor = 100
		  modifier = {
			  factor = 0
			  GER = { ai = no }
		  }
		}
		any_country = { limit = { war_with = GER }  end_war = GER }
		GER = { west_germany_region = { secede_province = USA } }
		
		# Dissolution of Nazi Germany
		inherit = GER
		any_owned = { limit = { is_core = GER }  remove_core = GER }
		any_country = { any_owned = { limit = { is_core = GER }  remove_core = GER } }
		
		### BRD cliams ###
		west_germany_region   = { add_core = DFR }
		east_germany_region   = { add_core = DFR } # DDR provinces
		polish_germany_region = { add_core = DFR } # Silesia-Pomerania
		polish_soviet_region  = { add_core = DFR } # Southern East Prussia
		north_germany_region  = { add_core = DFR } # Northern East Prussia
	}
	
	option = {
		name = "Restore the Weimar Republic"
		ai_chance = {
		  factor = 100
		  modifier = {
			  factor = 0
			  GER = { ai = yes }
		  }
		}
		
		set_global_flag = restore_weimar_republic
		any_country = { limit = { war_with = GER }  end_war = GER }
		USA = { create_vassal = GER }
		
		# Occupation
		GER = {
		  threat = { who = all value = -1000 }
		  any_owned = { change_controller = USA }
		  west_germany_region   = { remove_core = DFR }
		  country_event = 95352
		}
	}
}
